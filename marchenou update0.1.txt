        // Translate page content
        function translatePage(lang) {
            const trans = translations[lang] || translations['fr'];
            
            // Update navigation
            document.querySelectorAll('[data-translate]').forEach(el => {
                const key = el.getAttribute('data-translate');
                if (trans[key]) {
                    el.textContent = trans[key];
                }
            });
            
            // Update buttons and other elements
            document.querySelectorAll('[data-translate-attr]').forEach(el => {
                const [attr, key] = el.getAttribute('data-translate-attr').split(':');
                if (trans[key]) {
                    el.setAttribute(attr, trans[key]);
                }
            });
        }
